/** * Apps Script - שאר ישוב * לוגיקה מעודכנת: * 1. חיפוש 6 ספרות אחרונות בכל עמודה C. * 2. אם נמצא - עדכון השורה הקיימת. * 3. אם לא נמצא - הוספה החל משורה 853. * 4. כתיבת תאריך כטקסט למניעת היפוך (2.3 vs 3.2). */ function doPost(e) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const params = e.parameter; const fullPhone = params.id || ""; const status = params.status || ""; const guests = params.guests || 0; const family = params.family || 0; const timestamp = params.timestamp || ""; // הפורמט מה-HTML הוא D/M/YYYY if (!fullPhone) return ContentService.createTextOutput("Error: No ID"); const cleanInput = String(fullPhone).replace(/\D/g, ''); const last6 = cleanInput.length >= 6 ? cleanInput.slice(-6) : cleanInput; const values = sheet.getDataRange().getValues(); let targetRow = -1; // חיפוש גלובלי בכל הגיליון בעמודה C (אינדקס 2) for (let i = 1; i < values.length; i++) { let cellValue = String(values[i][2] || "").replace(/\D/g, ''); if (cellValue.length >= 6 && cellValue.slice(-6) === last6) { targetRow = i + 1; // נמצאה שורה קיימת break; } } // אם לא נמצא - קביעת שורה חדשה (מינימום 853) if (targetRow === -1) { let lastRowInSheet = sheet.getLastRow(); targetRow = Math.max(853, lastRowInSheet + 1); // כתיבת הטלפון המלא למשתמש חדש בעמודה C sheet.getRange(targetRow, 3).setValue(fullPhone); } // עדכון הנתונים // עמודה D (4): סטטוס // עמודה E (5): אורחים // עמודה F (6): משפחה // עמודה L (12): תאריך - נכתב עם גרש (') כדי שגוגל לא יהפוך אותו sheet.getRange(targetRow, 4).setValue(status); sheet.getRange(targetRow, 5).setValue(guests); sheet.getRange(targetRow, 6).setValue(family); // הכרחי: שימוש בגרש למניעת היפוך תאריך ע"י גוגל const safeDate = "'" + timestamp.replace(/\//g, '.'); sheet.getRange(targetRow, 12).setValue(safeDate); return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } catch (error) { return ContentService.createTextOutput("Error: " + error.toString()); } } function doGet(e) { return doPost(e); }
